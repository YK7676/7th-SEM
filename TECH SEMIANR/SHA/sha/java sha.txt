The Secure Hash Algorithm (SHA) is a widely used cryptographic hash function that is designed to produce a unique, fixed-size output for any input. Here is an example of how to implement SHA-256 in Java without using any libraries:

import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;

public class SHA256 {
    public static String hash(String input) {
        try {
            MessageDigest sha = MessageDigest.getInstance("SHA-256");
            byte[] digest = sha.digest(input.getBytes());
            StringBuilder sb = new StringBuilder();
            for (byte b : digest) {
                sb.append(String.format("%02x", b));
            }
            return sb.toString();
        } catch (NoSuchAlgorithmException e) {
            e.printStackTrace();
            return null;
        }
    }

    public static void main(String[] args) {
        String input = "Hello, World!";
        String output = hash(input);
        System.out.println("Input: " + input);
        System.out.println("Output: " + output);
    }
}
